INSTALLATION

cd my-project
npm install tailwindcss @tailwindcss/vite

Vite-Konfiguration (vite.config.ts)
import { defineConfig }
import tailwindcss from '@tailwindcss/vite'  

export default defineConfig({
  plugins: [tailwindcss()],
})
